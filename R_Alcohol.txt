max.pro <- function(drikke3){
  if(sum(is.na(drikke3)) == 3) return(NA)
  if(sum(is.na(drikke3)) < 3) return(max(drikke3, na.rm = rm))
}
sum.pro <- function(drikke3){
  if(sum(is.na(drikke3)) == 3) return(NA)
  if(sum(is.na(drikke3)) < 3) return(sum(drikke3, na.rm = rm))
}
